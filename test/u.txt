<script>
document.getElementById('predictionForm').addEventListener('submit', async function(e) {
    e.preventDefault();

    const data = {
        age: parseFloat(document.getElementById('age').value),
        sex: parseInt(document.getElementById('sex').value),
        resting_bp: parseFloat(document.getElementById('resting_bp').value),
        max_heart_rate: parseFloat(document.getElementById('max_heart_rate').value),
        chest_pain_type: parseInt(document.getElementById('chest_pain_type').value),
        resting_ecg: parseInt(document.getElementById('resting_ecg').value),
        st_depression: parseFloat(document.getElementById('st_depression').value),
        st_slope: parseInt(document.getElementById('st_slope').value),
        exercise_angina: parseInt(document.getElementById('exercise_angina').value),
        num_major_vessels: parseInt(document.getElementById('num_major_vessels').value),
        thalassemia: parseInt(document.getElementById('thalassemia').value)
    };

    const resultDiv = document.getElementById('result');
    resultDiv.innerHTML = "⏳ Predicting...";

    try {
        const response = await fetch('http://localhost:8000/predict', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(data)
        });

        if (!response.ok) {
            throw new Error(`Server error: ${response.status}`);
        }

        const result = await response.json();

        // Color for prediction
        const predClass = result.prediction === "YES" ? "prediction-yes" : "prediction-no";

        // Probability bar width
        const probPercent = Math.round(result.probability * 100);

        resultDiv.innerHTML = `
            <strong>Prediction:</strong> <span class="${predClass}">${result.prediction}</span><br>
            <strong>Probability:</strong> ${result.probability.toFixed(3)}
            <div class="probability-bar" style="width: ${probPercent}%;"></div>
            <strong>Normalized Data:</strong>
            <pre>${JSON.stringify(result.normalized_data, null, 2)}</pre>
        `;
    } catch (err) {
        resultDiv.innerHTML = `❌ Error: ${err.message}`;
    }
});
</script>
